Find Header Text VBA Code

지정된 경로의 엑셀 파일들을 순회하며 특정 시트의 특정 셀 값이 조건에 맞는지 검사하고 결과를 출력하는 매크로입니다.

Sub FindHeaderText() Dim index As Integer Dim tmpBook As Workbook Dim szFileList() As String Dim tarPath As String, fileExt As String, fileName As String tarPath = InputBox("경로를 입력하세요") fileName = Dir(tarPath & "\" & fileExt) While fileName <> "" index = index + 1 ReDim Preserve szFileList(1 To index) szFileList(index) = fileName fileName = Dir() Wend For index = 1 To UBound(szFileList) Workbooks.Open tarPath & "\" & szFileList(index) Set tmpBook = Workbooks(szFileList(index)) tmpBook.Worksheets("본문").Select If Cells(10, 3) <> "CFD_SYS" And Cells(10, 3) <> "CFD_SYS_TEST" Then Debug.Print "CFD SYS 없음: " & szFileList(index) End If tmpBook.Close savechanges:=False Next End Sub
