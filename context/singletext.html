폴더 내의 여러 엑셀 파일을 열어 '제개정이력' 시트의 날짜, 작성자, 부서 정보와 '표지' 시트의 버전을 일괄 업데이트하는 매크로입니다.

Sub reMakeHistory() Dim szFileList() As String Dim tarPath As String Dim fileExt As String Dim fileName As String Dim index As Integer Dim tmpBook As Workbook fileExt = "*.xlsx" index = 0 tarPath = InputBox("경로를 입력하세요") fileName = Dir(tarPath & "\" & fileExt) While fileName <> "" index = index + 1 ReDim Preserve szFileList(1 To index) szFileList(index) = fileName fileName = Dir() Wend For index = 1 To UBound(szFileList) Debug.Print "[" & szFileList(index) & ":[" & Dir(szFileList(index), vbDirectory) & "]" If Dir(szFileList(index), vbDirectory) <> "" Then Debug.Print szFileList(index) & " is directory pass !!!" GoTo Continue End If Workbooks.Open tarPath & "\" & szFileList(index) Set tmpBook = Workbooks(szFileList(index)) tmpBook.Worksheets("제개정이력").Select Cells(3, 3) = "2024-07-03" Cells(5, 6) = "전장품" Cells(6, 3) = Cells(5, 3) + 2 Cells(6, 5) = "김명광" Cells(6, 6) = "전장품" Cells(7, 3) = "2024-08-09" Cells(7, 5) = "김동일" Cells(7, 6) = "" Cells(8, 3) = "2024-08-16" If Cells(8, 5) = "" Then Cells(8, 5) = "심화섭" End If Cells(8, 6) = "조성" tmpBook.Worksheets("표지").Select Cells(9, 11) = "Ver. 0.9" tmpBook.Close savechanges:=True Continue: Next Debug.Print "== END ==" End Sub
