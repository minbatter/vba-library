File Copy VBA Code

원본 파일을 선택한 후, 엑셀 시트에 나열된 파일명 리스트를 바탕으로 지정된 경로에 복사본을 생성하는 매크로입니다.

Sub proc_filecopy() Dim intChoice As Integer Dim strPath As String Dim sRow As Integer, eRow As Integer, sCol As Integer Dim index As Integer Dim targetFileName As String Dim filenamePrefix As String filenamePrefix = "D:\ 테스트 관련 정기배치 통테 2차산출물\" Application.FileDialog(msoFileDialogOpen).AllowMultiSelect = False intChoice = Application.FileDialog(msoFileDialogOpen).Show If intChoice <> 0 Then strPath = Application.FileDialog(msoFileDialogOpen).SelectedItems(1) MsgBox strPath End If Sheets("Sheet1").Activate sRow = Selection.Row eRow = Selection.Row + Selection.Rows.Count - 1 sCol = Selection.Column For index = 1 To Selection.Rows.Count targetFileName = filenamePrefix & Cells(sRow + index - 1, sCol) VBA.FileCopy strPath, targetFileName Next End Sub
